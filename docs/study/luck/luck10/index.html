                    def solution(info, n, m):
    N = len(info)
    
    for _ in range(N):
        cur_m, cur_n, cur_N = m, n, len(info)
        A_flag, B_flag = False, False
        
        for i in range(cur_N):
            info.sort(key=lambda x : x[1])
            
            cur_a = info[i][0]
            cur_b = info[i][1]
            
            while not A_flag: 
                while not B_flag:
                    if cur_m > cur_b:
                        cur_m -= cur_b
