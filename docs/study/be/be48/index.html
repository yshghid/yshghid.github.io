                  from sentence_transformers import SentenceTransformer
import numpy as np

# 1лӢЁкі„: л¬ёмһҘ мһ„лІ л”©
model = SentenceTransformer('paraphrase-MiniLM-L6-v2')

reviews = [
    "л°°мҶЎмқҙ л№ лҘҙкі  м ңн’ҲлҸ„ мўӢм•„мҡ”.",
    "н’Ҳм§Ҳмқҙ кё°лҢҖ мқҙмғҒмһ…лӢҲлӢӨ!",
    "мғқк°ҒліҙлӢӨ л°°мҶЎмқҙ мҳӨлһҳ кұёл ём–ҙмҡ”.",
    "л°°мҶЎмқҖ лҠҗл ём§Җл§Ң нҸ¬мһҘмқҖ м•Ҳм „н–Ҳм–ҙмҡ”.",
    "м•„мЈј л§ҢмЎұмҠӨлҹ¬мҡҙ м ңн’Ҳмһ…лӢҲлӢӨ."
]

embeddings = model.encode(reviews)

                  import psycopg2
from pgvector.psycopg2 import register_vector

# 2лӢЁкі„: PostgreSQL н…Ңмқҙлё” мғқм„ұ
conn = psycopg2.connect(
    host="localhost",          
    port=5432,                
    database="postgres",    
    user="postgres",       
    password=os.getenv("PG_PASSWORD"),  # нҷҳкІҪліҖмҲҳм—җм„ң л¶Ҳлҹ¬мҳҙ 
)

cur = conn.cursor()

# DB мҙҲкё°нҷ” (кё°мЎҙ н…Ңмқҙлё” мӮӯм ң нӣ„ мһ¬мғқм„ұ)
dim = model.get_sentence_embedding_dimension()
cur.execute("CREATE EXTENSION IF NOT EXISTS vector;")
cur.execute("DROP TABLE IF EXISTS review_vectors;")  # н…Ңмқҙлё” мҷ„м „ мӮӯм ң
cur.execute(f"""
    CREATE TABLE review_vectors (
        id SERIAL PRIMARY KEY,
        review TEXT,
        embedding vector({dim})
    );
""")
conn.commit()

                  # 3лӢЁкі„: лІЎн„° м ҖмһҘ
for review, emb in zip(reviews, embeddings):

        (review, np.array(emb, dtype=np.float32)) 
    )
conn.commit()

                  # 4лӢЁкі„: мң мӮ¬лҸ„ кІҖмғү

for review, dist in cur.fetchall():
