                    from collections import deque

# ì…ë ¥
n, m, s = map(int, input().split())  # ì •ì  ê°œìˆ˜, ê°„ì„  ê°œìˆ˜, ì‹œì‘ ì •ì 
graph = {i: [] for i in range(1, n+1)}

# ê°„ì„  ì…ë ¥

    a, b = map(int, input().split())
    graph[a].append(b)
    graph[b].append(a)  # ì–‘ë°©í–¥ ê·¸ë˜í”„

# ì¸ì ‘ ë¦¬ìŠ¤íŠ¸ ì •ë ¬ (ì˜¤ë¦„ì°¨ìˆœ ë°©ë¬¸)
for node in graph:
    graph[node].sort()

# BFS í•¨ìˆ˜
def bfs(start):
    visited = set()
    queue = deque([start])
    visited_order = []

    while queue:
        v = queue.popleft()
        if v not in visited:
            visited.add(v)
            visited_order.append(v)
            for neighbor in graph[v]:
                if neighbor not in visited:
                    queue.append(neighbor)
    return visited_order

# ì¶œë ¥
