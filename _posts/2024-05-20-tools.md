### fastq 파일 길이 확인 Checking the length of fastq file
```python
import os

path='./Bowtie2Index'
fastq_list = [file for file in os.listdir(path) if file.endswith(".fastq")]
fastq_list.sort()

for fastq in fastq_list:
    fastq_path=path+f'/{fastq}'
    with open(fastq_path, 'r') as file:
        lines = [line.rstrip('\n') for line in file]
        print(fastq)
        print('read counts: ',(len(lines)+3)//4)
        for idx,i in enumerate(lines):  
            if (idx+1)%2==0 and len(i) !=101:
                print(f'line {idx+1}(4n+{(idx+1)%4})')
                print(f'length : {len(i)}')
```
