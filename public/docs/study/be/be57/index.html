    Github #4 .env 히스토리에서 제거 #

    #2026-02-08

    .env를 gitignore에 안넣고 잘못 Push 했을때.

    # .env를 git 히스토리에서 제거
grep -n '\.env' .gitignore 2>/dev/null || echo ".env not in .gitignore"

# 과거 커밋에 .env 제거
FILTER_BRANCH_SQUELCH_WARNING=1 git filter-branch --force --index-filter 'git rm --cached --ignore-unmatch .env' --prune-empty -- --all 2>&1

# stash를 복원하고 push
git stash pop 2>&1
git push -u origin main 2>&1
