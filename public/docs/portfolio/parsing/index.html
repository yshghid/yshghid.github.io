  // ìš°ì„ ìˆœìœ„ 10: Alt Text (ì´ë¯¸ì§€)

    static detectRole(element) {
  const roleMap = {
    'button': 'button',
    'a': 'link',
    'input': 'textbox',          // typeì— ë”°ë¼ ë³€ê²½
    'textarea': 'textbox',
    'select': 'combobox',
    'checkbox': 'checkbox',
    'radio': 'radio',
    'table': 'table',
    'img': 'img',
    'h1': 'heading',
    'h2': 'heading',
    'nav': 'navigation',
    'form': 'form'
  };

  // inputì€ typeì— ë”°ë¼ ì„¸ë¶„í™”
  if (element.type === 'input') {
    const inputTypes = {
      'checkbox': 'checkbox',
      'radio': 'radio',
      'email': 'textbox',
      'password': 'textbox',
      'text': 'textbox',
      'search': 'searchbox',
      'number': 'spinbutton'
    };
    return inputTypes[element.inputType] || 'textbox';
  }

  return roleMap[element.type] || null;
}
