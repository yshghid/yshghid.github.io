### 1. мӢӨмҠө1

<mark>мӢӨмҠө мӢңлӮҳлҰ¬мҳӨ</mark>

- мӮ¬мҡ©мһҗк°Җ м„Өкі„м•Ҳ н…ҚмҠӨнҠё(мҳҲ: description)лҘј мһ…л Ҙ
- н•ҙлӢ№ н…ҚмҠӨнҠём—җ лҢҖн•ҙ Pythonм—җм„ң AI мһ„лІ л”©мқ„ мҲҳн–ү

- мӢӨнҢЁн•ҳл©ҙ м•„л¬ҙ лҚ°мқҙн„°лҸ„ л“ұлЎқн•ҳм§Җ м•ҠмқҢ (ROLLBACK)

#2 python

```python
import psycopg2
import pandas as pd
import os
from dotenv import load_dotenv
from openai import OpenAI

os.chdir("/Users/yshmbid/Documents/home/github/SQL")

# 1. .env нҢҢмқј лЎңл“ң
load_dotenv()

# 2. OpenAI Client мғқм„ұ
client = OpenAI(api_key=os.getenv("OPENAI_API_KEY")) # .envм—җм„ң OPENAI_API_KEY к°Җм ёмҳҙ

# 3. PostgreSQL DBм—җ м—°кІ°
conn = psycopg2.connect(
    host="localhost",
    port=5432,
    database="postgres",
    user="postgres",
    password=os.getenv("PG_PASSWORD"), # .envм—җм„ң PG_PASSWORD к°Җм ёмҳҙ
)
cursor = conn.cursor() # SQLл¬ё мӢӨн–ү

# 4. мһ„лІ л”© н•ЁмҲҳ
def get_embedding(text: str):
    response = client.embeddings.create(
        input=text,
        model="text-embedding-3-small"  # вҖңtext-embedding-3-smallвҖқ лӘЁлҚёлЎң мһ„лІ л”© мғқм„ұ
    )
    return response.data[0].embedding 

# 5. DB мӮҪмһ… н•ЁмҲҳ
def insert_design(description: str):
    try:
        cursor.execute("BEGIN;")  # нҠёлһңмһӯм…ҳ мӢңмһ‘

        # мһ„лІ л”© мғқм„ұ
        embedding = get_embedding(description)

        # design н…Ңмқҙлё”м—җ мӮҪмһ…
        cursor.execute(
            "INSERT INTO design (description, embedding) VALUES (%s, %s)",
            (description, embedding)
        )

        conn.commit()
        print(f"[COMMIT] л“ұлЎқ м„ұкіө вҶ’ {description[:40]}...")

    except Exception as e:
        conn.rollback()
        print(f"[ROLLBACK] мӢӨнҢЁ вҶ’ {description[:40]}... м—җлҹ¬: {e}")

# 6. CSV нҢҢмқј лЎңл“ң & мІҳлҰ¬
df = pd.read_csv("sample_designs_500.csv")

for idx, row in df.iterrows():
    desc = row.get("description")
    if pd.notna(desc):  # descriptionмқҙ л№„м–ҙмһҲм§Җ м•Ҡмқ„ л•Ңл§Ң мӢӨн–ү
        insert_design(desc)

# 7. м—°кІ° мў…лЈҢ
cursor.close()
conn.close()
```

#4 к°ңл…җ

нҠёлһңм қм…ҳ?

#1 FastAPI м„ңлІ„

```python
from fastapi import FastAPI, HTTPException
from pydantic import BaseModel
import psycopg2
from dotenv import load_dotenv
from openai import OpenAI
import os

# кІҪлЎң м„Өм •
os.chdir("/Users/yshmbid/Documents/home/github/SQL")

# .env лЎңл“ң
load_dotenv()

# OpenAI нҒҙлқјмқҙм–ёнҠё
client = OpenAI(api_key=os.getenv("OPENAI_API_KEY"))

# DB м—°кІ°
conn = psycopg2.connect(
    host="localhost",
    port=5432,
    database="postgres",
    user="postgres",
    password=os.getenv("PG_PASSWORD"),
)
cursor = conn.cursor()

# FastAPI м•ұ к°қмІҙ мғқм„ұ
app = FastAPI()

# мҡ”мІӯ лҚ°мқҙн„° лӘЁлҚё м •мқҳ
class DesignRequest(BaseModel):
    description: str

# мһ„лІ л”© н•ЁмҲҳ
def get_embedding(text: str):
    response = client.embeddings.create(
        model="text-embedding-3-small",
        input=text
    )
    return response.data[0].embedding

@app.post("/register_design")
def register_design(req: DesignRequest):
    try:
        cursor.execute("BEGIN;")
        embedding = get_embedding(req.description)

        cursor.execute(
            "INSERT INTO design (description, embedding) VALUES (%s, %s)",
            (req.description, embedding)
        )
        conn.commit()
        return {"status": "success", "message": "л“ұлЎқ м„ұкіө"}
    except Exception as e:
        conn.rollback()
        raise HTTPException(status_code=500, detail=f"л“ұлЎқ мӢӨнҢЁ: {e}")
```

class DesignRequestмҷҖ register_design()мҷҖмқҳ нҳёнҷҳ?

### 3. мӢӨмҠө2 - л ҲнҚјлҹ°мҠӨ мҪ”л“ң

```python 
# AI м„Өкі„м•Ҳ л“ұлЎқ нҶөн•© мӢӨмҠө (FastAPI + Streamlit + PostgreSQL)

## FastAPI м„ңлІ„ (app.py)
from fastapi import FastAPI, HTTPException
from pydantic import BaseModel
from sentence_transformers import SentenceTransformer
import psycopg2

app = FastAPI()
model = SentenceTransformer('all-MiniLM-L6-v2')

def get_db_conn():
    return psycopg2.connect(
        dbname="yourdb", user="youruser", password="yourpass", host="localhost"
    )

class DesignInput(BaseModel):
    title: str
    description: str

@app.post("/register_design")
def register_design(data: DesignInput):
    conn = get_db_conn()
    cur = conn.cursor()
    try:
        embedding = model.encode(data.description).tolist()
        cur.execute("BEGIN;")
        cur.execute("""
            INSERT INTO design (title, description, embedding)
            VALUES (%s, %s, %s);
        """, (data.title, data.description, embedding))
        conn.commit()
        return {"status": "success", "message": "л“ұлЎқ мҷ„лЈҢ"}
    except Exception as e:
        conn.rollback()
        raise HTTPException(status_code=500, detail=f"л“ұлЎқ мӢӨнҢЁ: {str(e)}")
    finally:
        cur.close()
        conn.close()
```
