# env #3 venv 초기 설정법

#2026-05-26

---

```bash
# 1. 가상환경 생성
python3.11 -m venv .venv

# 2. 활성화
source .venv/bin/activate

# 3. pip 업그레이드
pip install --upgrade pip

# 4. 패키지 설치
pip install -r requirements.txt

# 5. typing_extensions 업그레이드 (IPython 호환성)
pip install "typing_extensions>=4.6.0" --upgrade

# 6. Jupyter kernel 등록
python -m ipykernel install --user --name workspace --display-name "Python (workspace)"

# 7. 노트북 실행
jupyter lab
```
