#patients.bash

```bash
#!/bin/bash

# Input:
# Output:
# 1) TXT file with patient id

ALLPROT_PATH="/data/c315/allprot.fasta"
OUT_DIR="/data"

# Go to the script directory and load config file
cd $(dirname $0)
#source ../config.bash

# Initialize the patient list file without adding an empty line
> "$OUT_DIR/patient_id.txt" 

while read -r line
do
  if [[ $line == ">"* ]]; then
    PATIENT_ID=$(echo $line | cut -d'|' -f1 | tr -d '>')
    echo "$PATIENT_ID" >> "$OUT_DIR/patient_id.txt"
  fi
done < "$ALLPROT_PATH"
```
